//-------------------------Replace Characters

var clean = [/</ig, />/ig, /[/]/ig, /[=]/ig, /["]/ig, /[!]/ig, /[-]/ig, /[\t]/ig];
var rep = ['<', '>', '/', '=', '"', '!', '-', '\s\s\s\s'];

//--------------------------------------------------HTML
//-------------------------Fix
//Fixes the colouring of all the > characters
//at the comments so that the > character
//at all the comments are the same colour
var htmlFix = /><\/span><\/span>/ig;
//-------------------------Comment
var htmlCom = /(<!DOCTYPE|<!--)([\s\S]*?)(-->|>)/ig;
//-------------------------Tag
var htmlTag = /(<|</)([\w]+)|>/ig;
//-------------------------Attribute
var htmlAtt = /([\w]+)=/ig;
//-------------------------Value
var htmlVal = /"([\s\S]*?)"/ig;
//--------------------------------------------------CSS
//-------------------------Fix
//Fixes the colouring of all the : characters
//at the comments so that the : character
//at all the comments are the same colour
var cssFixa = /:<\/span>/ig;
var cssFixb = /:<\/span>/ig;
//Fixes the colouring of all the { characters
//at the comments so that the { character
//at all the comments are the same colour
var cssFixc = /{<\/span><\/span>/ig;
//-------------------------Comment
var cssCom = //\*([\s\S]*?)\*//ig;
//-------------------------Selector
var cssSel = /([\w\s.#:_@!\[\]\(\)&45;]*?){|}/ig;
//-------------------------Selector Extention
var cssSelExt = /:(.*?){/ig;
//-------------------------Property
var cssProp = /([\w-]*?):/ig;
//-------------------------Value
var cssVal = /:(.*?);/ig;
//-------------------------Unit
var cssUnt = /([^\D])([\d.]*?)(em|ex|%|px|cm|mm|in|pt|pc|ch|rem|vh|vw|vmin|vmax)/ig;

//---------------------------------------------------------------------------Colour Code
function colourCode() {
  //--------------------------------------------------HTML
  $('#html').each(function() {
    //-------------------------Get Text
    var htmlStr = $(this).html();
    //-------------------------Replace Characters
    for (a = 0; a < clean.length; a++) {
      htmlStr = htmlStr.replace(clean[a], rep[a]);
    }
    //-------------------------Wrap Matching Text
    htmlStr = htmlStr.replace(htmlCom, '$&');
    htmlStr = htmlStr.replace(htmlTag, '$&');
    htmlStr = htmlStr.replace(htmlAtt, '$&');
    htmlStr = htmlStr.replace(htmlVal, '$&');
    htmlStr = htmlStr.replace(htmlFix, '>');
    //-------------------------Numbering
    //Doesn't work that well. Update should fix it. 
    //htmlStr = htmlStr.replace(/<!--([\s\S]*?)\n/ig, '$&');
    //htmlStr = htmlStr.replace(/<\/span>

    //htmlStr = htmlStr.replace(/([\s\S]+)/ig, '
    1. $&
    ');
    //-------------------------Insert Coloured Text
    $(this).html(htmlStr);
  });
  //--------------------------------------------------CSS
  $('#css').each(function() {
    //-------------------------Get Text
    var cssStr = $(this).html();
    //-------------------------Replace Characters
    cssStr = cssStr.replace(/[;]/ig, ';');
    for (b = 0; b < clean.length; b++) {
      cssStr = cssStr.replace(clean[b], rep[b]);
    }
    //-------------------------Wrap Matching Text
    cssStr = cssStr.replace(cssCom, '$&');
    cssStr = cssStr.replace(cssSel, '$&');
    cssStr = cssStr.replace(cssSelExt, '$&');
    cssStr = cssStr.replace(cssProp, '$&');
    cssStr = cssStr.replace(cssVal, '$&');
    cssStr = cssStr.replace(cssUnt, '$&');
    cssStr = cssStr.replace(cssFixa, ':');
    cssStr = cssStr.replace(cssFixb, ':');
    cssStr = cssStr.replace(cssFixc, '{');
    //-------------------------Insert Coloured Text
    $(this).html(cssStr);
  });
}
